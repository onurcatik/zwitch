Introduction

MERN Stack ile Twitch Klon Uygulaması Oluşturma

Genel Bakış

Bu eğitimde, MERN stack (MongoDB, Express.js, React.js ve Node.js) kullanarak bir Twitch klon uygulaması oluşturacağız. Bu uygulama, kanal tarama, akış izleme, sohbet özellikleri, kullanıcı kimlik doğrulama ve bildirimler gibi Twitch'in temel işlevlerinin çoğunu kopyalayacaktır. Uygulamanın uygulanmasının kesin, titiz ve yazılım geliştirme alanındaki en iyi uygulamalara uygun olmasını sağlayacağız.

İşlevler

  1. Kanal Tarama: Kullanıcılar bir dizi kanalı tarayabilir.
  2. Kanal Detayları: Her kanalın avatarını, başlığını, açıklamasını ve çevrimiçi durumunu görüntüleyebilir.
  3. Kanal İzleme: Kullanıcılar, kanal çevrimiçi ise akışları izleyebilir, kanalın açıklamasını ve takip etme düğmesini görebilir.
  4. Sohbet Sistemi: Her kanalın bir sohbet özelliği olacak ve ziyaretler arasında mesajlar saklanacaktır.
  5. Kullanıcı Kimlik Doğrulama: Kullanıcılar kaydolabilir, giriş yapabilir ve kimlik doğrulaması yapılırsa kullanıcı adlarını görebilir.
  6. Kanal Yönetimi: Kullanıcılar, kullanıcı adlarını, başlıklarını, avatarlarını ve açıklamalarını ayarlayarak kanallarını yönetebilir.
  7. Yayın Entegrasyonu: Her kullanıcının, OBS gibi yayın yazılımına bağlanmak için benzersiz bir yayın anahtarı olacaktır.
  8. Bildirimler: Kullanıcılar, örneğin yanlış giriş kimlik bilgileri gibi hatalar için bildirimler alacaklar.

Ön Koşullar

Başlamadan önce, aşağıdakilerin yüklü olduğundan emin olun:

Kullanıcı Kimlik Doğrulama

Backend

Bir models dizini ve kullanıcı şeması için bir User.js dosyası oluşturun.

const mongoose = require('mongoose');
const bcrypt = require('bcryptjs');

const UserSchema = new mongoose.Schema({
    username: { type: String, required: true, unique: true },
    email: { type: String, required: true, unique: true },
    password: { type: String, required: true },
});

// Kaydetmeden önce şifreyi hashleyin
UserSchema.pre('save', async function(next) {
    if (!this.isModified('password')) {
        return next();
    }
    const salt = await bcrypt.genSalt(10);
    this.password = await bcrypt.hash(this.password, salt);
    next();
});

module.exports = mongoose.model('User', UserSchema);

Kullanıcı kaydı ve giriş için bir routes dizini oluşturun.

const express = require('express');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const User = require('../models/User');

const router = express.Router();

// Register
router.post('/register', async (req, res) => {
    const { username, email, password } = req.body;

    try {
        const user = new User({ username, email, password });
        await user.save();
        res.status(201).send('User registered');
    } catch (error) {
        res.status(400).send(error.message);
    }
});

// Login
router.post('/login', async (req, res) => {
    const { email, password } = req.body;

    try {
        const user = await User.findOne({ email });
        if (!user) {
            return res.status(400).send('Invalid credentials');
        }

        const isMatch = await bcrypt.compare(password, user.password);
        if (!isMatch) {
            return res.status(400).send('Invalid credentials');
        }

        const token = jwt.sign({ userId: user._id }, 'your_jwt_secret', { expiresIn: '1h' });
        res.json({ token });
    } catch (error) {
        res.status(500).send('Server error');
    }
});

module.exports = router;

Bu route'ları server.js dosyasına entegre edin.

const userRoutes = require('./routes/users');
app.use('/api/users', userRoutes);

// LoginForm.js
import React, { useState } from 'react';
import axios from 'axios';

const LoginForm = () => {
    const [email, setEmail] = useState('');
    const [password, setPassword] = useState('');
    const [error, setError] = useState('');

    const handleSubmit = async (e) => {
        e.preventDefault();
        try {
            const res = await axios.post('http://localhost:5000/api/users/login', { email, password });
            localStorage.setItem('token', res.data.token);
            setError('');
        } catch (err) {
            setError('Invalid credentials');
        }
    };

    return (
        <form onSubmit={handleSubmit}>
            <input type="email" value={email} onChange={(e) => setEmail(e.target.value)} placeholder="Email" required />
            <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} placeholder="Password" required />
            <button type="submit">Login</button>
            {error && <p>{error}</p>}
        </form>
    );
};

export default LoginForm;

// Chat.js
import React, { useState, useEffect } from 'react';
import axios from 'axios';

const Chat = ({ channelId, userId }) => {
    const [messages, setMessages] = useState([]);
    const [message, setMessage] = useState('');

    useEffect(() => {
        const fetchMessages = async () => {
            const res = await axios.get(`http://localhost:5000/api/chat/${channelId}`);
            setMessages(res.data);
        };

        fetchMessages();
    }, [channelId]);

    const handleSendMessage = async (e) => {
        e.preventDefault();
        if (message.trim()) {
            await axios.post(`http://localhost:5000/api/chat/${channelId}`, { userId, message });
            setMessage('');
            const res = await axios.get(`http://localhost:5000/api/chat/${channelId}`);
            setMessages(res.data);
        }
    };

    return (
        <div>
            <div>
                {messages.map((msg) => (
                    <p key={msg._id}><strong>{msg.user.username}</strong>: {msg.message}</p>
                ))}
            </div>
            <form onSubmit={handleSendMessage}>
                <input type="text" value={message} onChange={(e) => setMessage(e.target.value)} placeholder="Type a message" required />
                <button type="submit">Send</button>
            </form>
        </div>
    );
};

export default Chat;

Socket.io ile MERN Stack

Teknolojilere Giriş

Bu eğitimde, MERN stack'e derinlemesine bakacağız. MERN, MongoDB, Express.js, React.js ve Node.js anlamına gelir. Ek olarak, gerçek zamanlı sohbet işlevselliği için Socket.io'yu entegre edeceğiz. Bu teknolojilerin her biri, tam özellikli bir web uygulaması oluşturmak için kritik bir rol oynamaktadır. Her bileşene kısa bir giriş yaparak başlayalım.

MongoDB

MongoDB, verileri esnek, JSON benzeri bir formatta saklayan bir NoSQL veritabanıdır. Modern web uygulamalarının büyük veri hacimlerini işleyebilmesi için ölçeklenebilirlik ve esneklik sunar.

Express.js

Express.js, Node.js için bir web uygulama framework'üdür. API'ler oluşturmayı basitleştiren bir dizi middleware bileşeni sağlayarak web ve mobil uygulamalar geliştirmek için güçlü bir özellik seti sunar.

React.js

React.js, kullanıcı arayüzleri oluşturmak için bir JavaScript kütüphanesidir. Yeniden kullanılabilir UI bileşenleri oluşturmayı, karmaşık uygulamaların durumunu yönetmeyi ve veri değişikliklerinde doğru bileşenleri verimli bir şekilde güncelleyip render etmeyi sağlar.

Node.js

Node.js, Chrome'un V8 JavaScript motoru üzerine inşa edilmiş bir JavaScript runtime'dır. JavaScript ile sunucu tarafında ölçeklenebilir ağ uygulamaları oluşturmayı sağlar.

Socket.io

Socket.io, web istemcileri ve sunucuları arasında gerçek zamanlı, çift yönlü ve olay tabanlı iletişim sağlayan bir kütüphanedir. Genellikle sohbet uygulamaları, gerçek zamanlı analizler ve işbirliği araçları oluşturmak için kullanılır.

MERN Stack Uygulamasının Mimarisi

MERN stack uygulamamızın mimarisi şu bileşenlerden oluşacaktır:

  1. İstemci Tarafı (Frontend)

    • Frontend, React.js kullanılarak oluşturulacaktır. Verileri almak ve görüntülemek için backend ile API çağrıları yapacaktır.

  2. Sunucu Tarafı (Backend)

    • Backend, Node.js ve Express.js kullanılarak oluşturulacaktır. İstemci isteklerini işleyip veritabanı ile etkileşime geçecek ve API uç noktalarını sunacaktır.
    • Uygulamanın verilerini depolayıp yönetmek için MongoDB kullanılacaktır.
    • Sohbet işlevselliği gibi gerçek zamanlı iletişim sağlamak için sunucuya Socket.io entegre edilecektir.

import React, { useState, useEffect } from 'react';

const socket = io('http://localhost:3001');

function Chat() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');

  useEffect(() => {
    socket.on('chat message', (msg) => {
      setMessages([...messages, msg]);
    });
  }, [messages]);

  const sendMessage = () => {
    socket.emit('chat message', input);
    setInput('');
  };

  return (
    <div>
      <ul>
        {messages.map((msg, index) => (
          <li key={index}>{msg}</li>
        ))}
      </ul>
      <input
        type="text"
        value={input}
        onChange={(e) => setInput(e.target.value)}
      />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}

export default Chat;

Bileşenler

Bileşenler, bir React uygulamasının yapı taşlarıdır. İki tür bileşen vardır:

  • Functional Components: JSX döndüren basit JavaScript fonksiyonlarıdır. Props kabul edebilirler ve state ve yaşam döngüsü metodları için hooks kullanırlar.
  • Class Components: React.Componenti genişleten ve JSX döndürmek için render metoduna sahip ES6 sınıflarıdır. Class bileşenleri kendi state'lerini yönetebilir ve yaşam döngüsü metodlarına sahiptir.

Functional Component Örneği:

import React from 'react';

function Greeting(props) {
  return <h1>Hello, {props.name}!</h1>;
}

export default Greeting;

Class Component Örneği:

import React, { Component } from 'react';

class Greeting extends Component {
  render() {
    return <h1>Hello, {this.props.name}!</h1>;
  }
}

export default Greeting;

Props ve State

  • Props: Props (properties'in kısaltması), parent bileşenlerden child bileşenlere geçirilen salt okunur özelliklerdir. Verileri ve olay işleyicilerini child bileşenlere iletmek için kullanılırlar.

  • State: State, bileşene özel olan ve zamanla değişebilen yerel veri deposudur. Bir bileşenin state'i değiştiğinde, bileşen yeni state'i yansıtmak için yeniden render edilir.

Class Component'te State Kullanımı Örneği:

import React, { Component } from 'react';

class Counter extends Component {
  constructor(props) {
    super(props);
    this.state = { count: 0 };
  }

  increment = () => {
    this.setState({ count: this.state.count + 1 });
  }

  render() {
    return (
      <div>
        <p>Count: {this.state.count}</p>
        <button onClick={this.increment}>Increment</button>
      </div>
    );
  }
}

export default Counter;

Hooks ile Functional Component'te State Kullanımı Örneği:

import React, { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
}

export default Counter;

React ile Frontend Geliştirme

React, özellikle tek sayfa uygulamalarda, reaktivite ve performansın kritik olduğu kullanıcı arayüzleri oluşturmak için popüler bir JavaScript kütüphanesidir. React’in sanal DOM’u, verimli güncellemeler ve sorunsuz bir kullanıcı deneyimi sağlayarak render işlemlerini optimize eder. Ayrıca, Redux gibi durum yönetimi ve React Router gibi navigasyon için kullanılan geniş React kütüphaneleri ve araçları ekosistemi, yeteneklerini daha da artırır. Node.js ve Express ile Backend Geliştirme

Node.js, geliştiricilerin JavaScript’i sunucu tarafında çalıştırmalarına olanak tanıyan bir çalışma ortamıdır. Bu, stack boyunca geliştirme dilinin birleştirilmesini sağlar ve öğrenme eğrisini azaltır. Node.js, V8 JavaScript motoru üzerine kuruludur ve bu da onu hızlı ve verimli kılar, özellikle I/O yoğun işlemler için.

Express.js, Node.js için minimal bir web framework'üdür ve web uygulamaları ve API'ler oluşturmak için sağlam özellikler sunar. Routing, middleware entegrasyonu ve istek yönetimini basitleştirir. Express’in esnekliği, geliştiricilerin uygulamalarını ihtiyaçlarına göre yapılandırmalarına olanak tanır ve hem küçük hem de büyük ölçekli uygulamalar için uygundur.

Node.js ve Express.js, npm (Node Package Manager) aracılığıyla sunulan geniş bir middleware ve modül ekosistemine sahiptir. Bu geniş paket kütüphanesi, kimlik doğrulama, giriş doğrulama ve dosya işlemleri gibi çeşitli işlevleri hızla uygulamayı sağlar, tekerleği yeniden icat etmeye gerek kalmadan.

MongoDB ile Veritabanı Entegrasyonu

MongoDB’nin döküman odaklı yapısı, JavaScript ve JSON ile iyi uyum sağlar, Node.js ve Express ile kesintisiz entegrasyon sağlar.

MongoDB’nin en önemli avantajlarından biri ölçeklenebilirliğidir. Sharding aracılığıyla yatay ölçeklemeyi destekler ve verilerin birden fazla sunucuya dağıtılmasına olanak tanır. Bu özellik, uygulamanın artan yükü ve büyüyen veri hacimlerini performanstan ödün vermeden yönetmesini sağlar.

MongoDB ayrıca güçlü sorgulama ve indeksleme yetenekleri sunar, verilerin verimli bir şekilde alınmasını ve işlenmesini sağlar. Aggregation framework'ü, karmaşık veri dönüşümleri ve analitik işlemleri destekler, bu da onu geniş bir uygulama yelpazesi için uygun hale getirir.

const mongoose = require('mongoose');
const socketio = require('socket.io');
const http = require('http');

dotenv.config();

const server = http.createServer(app);
const io = socketio(server);

// Middleware

app.use(express.json());

// Temel route
app.get('/', (req, res) => {
  res.send('Server is running');
});

// Socket.io bağlantısı
io.on('connection', (socket) => {
  console.log('New WebSocket connection');
});

const PORT = process.env.PORT || 5000;

mongoose.connect(process.env.MONGO_URI, {
  useNewUrlParser: true,
  useUnifiedTopology: true,
}).then(() => {
  server.listen(PORT, () => console.log(`Server running on port ${PORT}`));
}).catch((error) => {
  console.error('Database connection error:', error);

import http from 'http';
import cors from 'cors';
import dotenv from 'dotenv';

  console.log(`Server is listening on port ${port}`);

  4. Temel Route: Basit bir GET route ayarlıyoruz ve bir mesajla yanıt veriyoruz.

    const { username, email, password } = req.body;

    try {
        let user = await User.findOne({ email });
        if (user) {

            email,
            password,

            },

Implementing JWT Authentication

JWT Kullanarak Kullanıcı Kaydı ve Kimlik Doğrulama

Giriş

Bu eğitimde, bir MERN stack uygulamasında kullanıcı kaydı ve kimlik doğrulamayı sistematik olarak uygulayacağız. Bu kılavuz, kullanıcı verilerini bir veritabanında saklamayı, benzersiz kullanıcı e-postalarını sağlamayı, hataları yönetmeyi ve kimlik doğrulama için JSON Web Tokens (JWT) kullanmayı kapsayacaktır.

Gereksinimler

  • MERN stack (MongoDB, Express.js, React.js, Node.js) hakkında temel bilgi
  • Node.js ve Express.js hakkında bilgi sahibi olmak
  • JavaScript'te asenkron programlamayı anlama (Promises, async/await)
  • JWT (JSON Web Tokens) hakkında temel bilgi

Uygulama Akışı

  1. Kullanıcı kaydı: İstemci, kullanıcı verilerini (kullanıcı adı, e-posta, parola) sunucuya gönderir.
  2. Sunucu, e-postanın benzersiz olup olmadığını kontrol eder.
  3. E-posta benzersizse, sunucu parolayı şifreler ve kullanıcıyı veritabanına kaydeder.
  4. Kimlik doğrulama ve yetkilendirme için JWT uygulanır.

Adım 1: Sunucuyu Kurma

1.1. Bağımlılıkları Yükleme

İlk olarak, yeni bir Node.js projesi kurun ve gerekli bağımlılıkları yükleyin:

npm install express mongoose bcryptjs jsonwebtoken body-parser

1.2. Express Sunucusunu Oluşturma

server.js dosyasını oluşturun ve temel sunucu yapısını kurun:

const express = require('express');
const mongoose = require('mongoose');
const bodyParser = require('body-parser');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

const app = express();

// Middleware
app.use(bodyParser.json());

// MongoDB bağlantısı
mongoose.connect('mongodb://localhost:27017/mern-auth', { useNewUrlParser: true, useUnifiedTopology: true })
    .then(() => console.log('MongoDB bağlandı'))
    .catch(err => console.log(err));

// Kullanıcı Şeması
const userSchema = new mongoose.Schema({
    username: { type: String, required: true },
    email: { type: String, required: true, unique: true },
    password: { type: String, required: true },
});

const User = mongoose.model('User', userSchema);

// Rotalar
app.post('/register', async (req, res) => {
    try {
        const { username, email, password } = req.body;
        
        // E-postanın zaten var olup olmadığını kontrol et
        const existingUser = await User.exists({ email });
        if (existingUser) {
            return res.status(409).json({ error: 'Email already in use' });
        }

        // Parolayı şifrele
        const hashedPassword = await bcrypt.hash(password, 10);

        // Kullanıcıyı veritabanına kaydet
        const newUser = new User({ username, email, password: hashedPassword });
        await newUser.save();

        res.status(201).json({ message: 'User registered successfully' });
    } catch (error) {
        res.status(500).json({ error: 'An error occurred, please try again' });
    }
});

// Sunucuyu başlat
const PORT = process.env.PORT || 5000;
app.listen(PORT, () => console.log(`Sunucu ${PORT} portunda çalışıyor`));

Adım 3: JWT ile Kimlik Doğrulama Uygulama

3.1. JWT Oluşturma

Bir kullanıcı giriş yaptığında, istemcinin korunan kaynaklara erişmek için kullanabileceği bir JWT oluştururuz.

3.2. Giriş Rotası Ekleme

app.post('/login', async (req, res) => {
    try {
        const { email, password } = req.body;
        
        // Kullanıcının var olup olmadığını kontrol et
        const user = await User.findOne({ email });
        if (!user) {
            return res.status(404).json({ error: 'User not found' });
        }

        // Parolayı doğrula
        const isMatch = await bcrypt.compare(password, user.password);
        if (!isMatch) {
            return res.status(400).json({ error: 'Invalid credentials' });
        }

        // JWT oluştur ve gönder
        const token = jwt.sign({ userId: user._id }, 'your_jwt_secret', { expiresIn: '1h' });
        res.status(200).json({ token });
    } catch (error) {
        res.status(500).json({ error: 'An error occurred, please try again' });
    }
});

3.3. Rotaları Koruma

Belirli rotaları korumak için JWT'yi doğrulayan middleware oluşturun:

const authMiddleware = (req, res, next) => {
    const token = req.header('Authorization').replace('Bearer ', '');
    if (!token) {
        return res.status(401).json({ error: 'Access denied, no token provided' });
    }

    try {
        const decoded = jwt.verify(token, 'your_jwt_secret');
        req.user = decoded;
        next();
    } catch (error) {
        res.status(401).json({ error: 'Invalid token' });
    }
};

// Korunan rota örneği
app.get('/protected', authMiddleware, (req, res) => {
    res.status(200).json({ message: 'Bu korunan bir rotadır' });
});

Bir kullanıcı kaydolduğunda, şifresinin güvenli bir şekilde saklanması çok önemlidir. Düz metin şifrelerin saklanması büyük bir güvenlik riskidir. Bunun yerine, şifreyi bcryptjs kullanarak hashleyeceğiz.

Gerekli Modüllerin İçe Aktarılması

Öncelikle, controller dosyanızda gerekli modülleri içe aktarın:

const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const User = require('../models/User'); // Gerektiği şekilde yolu ayarlayın

Register Controller Mantığı

Kayıt controller'ında, şifreyi hashlemek ve kullanıcı detaylarını veritabanına kaydetmek için mantığı uygulayın.

exports.registerUser = async (req, res) => {
    try {
        const { username, email, password } = req.body;

        // Kullanıcının zaten mevcut olup olmadığını kontrol edin
        const existingUser = await User.findOne({ email });
        if (existingUser) {
            return res.status(400).json({ message: "User already exists" });
        }

        // Şifreyi hashleyin
        const saltRounds = 10;
        const hashedPassword = await bcrypt.hash(password, saltRounds);

        // Yeni bir kullanıcı oluşturun
        const newUser = new User({
            username,
            email: email.toLowerCase(), // E-postanın küçük harflerle saklandığından emin olun
            password: hashedPassword
        });

        // Kullanıcıyı veritabanına kaydedin
        const savedUser = await newUser.save();

        // Bir JWT token oluşturun
        const token = jwt.sign(
            { userId: savedUser._id, email: savedUser.email },
            process.env.JWT_SECRET, // Güvenli ve özel bir anahtar kullanın
            { expiresIn: '1h' }
        );

        // Başarılı yanıt gönderin
        res.status(201).json({
            user: {
                id: savedUser._id,
                username: savedUser.username,
                email: savedUser.email
            },
            token
        });
    } catch (error) {
        res.status(500).json({ message: "Server error" });
    }
};

Açıklama

  1. Kullanıcı Varlık Kontrolü: Yeni bir kullanıcı oluşturmadan önce, sağlanan e-postayla zaten bir kullanıcı olup olmadığını veritabanında kontrol edin. Bu, tekrar kayıtların önlenmesini sağlar.

  2. Şifre Hashleme: Kullanıcının şifresini bcrypt.hash() kullanarak hashleyin. saltRounds parametresi, hashleme işleminin hesaplama karmaşıklığını belirler. Genellikle 10 değeri kullanılır.

  3. Kullanıcı Oluşturma: Hashlenmiş şifre ve diğer sağlanan detaylarla yeni bir kullanıcı nesnesi oluşturun. E-postayı tutarlılığı sağlamak için küçük harfe çevirin.

  4. JWT Oluşturma: Kullanıcıyı kaydettikten sonra, kullanıcının kimlik doğrulaması için kullanıcı ID ve e-posta içeren bir JWT token oluşturun.

  5. Yanıt: Kullanıcı detayları ve JWT token içeren bir yanıt gönderin.

Adım 3: JWT Kimlik Doğrulama İşlemi

JWT tokenları, korumalı yollar için kullanıcıları doğrulamak amacıyla kullanılır. Aşağıdaki middleware fonksiyonu tokenı doğrular:

JWT Middleware

Bir middleware dosyası oluşturun, örneğin authMiddleware.js:

const jwt = require('jsonwebtoken');

exports.verifyToken = (req, res, next) => {
    const token = req.headers['authorization'];

    if (!token) {
        return res.status(403).json({ message: "No token provided" });
    }

    jwt.verify(token, process.env.JWT_SECRET, (err, decoded) => {
        if (err) {
            return res.status(500).json({ message: "Failed to authenticate token" });
        }

        // Diğer yollar için decoded bilgileri request'e kaydedin
        req.userId = decoded.userId;
        next();
    });
};

Açıklama

  1. Token Çıkarımı: Tokenı request başlıklarından çıkarın.

  2. Token Doğrulama: Tokenı jwt.verify() kullanarak doğrulayın. Eğer token geçersiz veya süresi dolmuşsa, hata ile yanıt verin.

  3. Request Geliştirme: Eğer token geçerliyse, çözümlenmiş kullanıcı kimliğini diğer middleware veya route handler'larında kullanılmak üzere request nesnesine kaydedin.

Adım 4: Yolları Korumak

Kimlik doğrulama gerektiren yolları korumak için middleware'i kullanın. Örneğin:

const express = require('express');
const { verifyToken } = require('../middleware/authMiddleware');
const router = express.Router();

router.get('/protected', verifyToken, (req, res) => {
    res.status(200).json({ message: "Access granted to protected route" });
});

module.exports = router;

JSON Web Token (JWT) Oluşturma ve Kullanımı

1. jsonwebtoken Paketini Kurun

Öncelikle, jsonwebtoken paketini kurun. Terminalinizi açın ve şu komutu çalıştırın:

npm install jsonwebtoken

2. jsonwebtoken Paketini İçe Aktarın

postRegister.js dosyanızda (veya kullanıcı kaydı işlemini gerçekleştiren herhangi bir dosyada) jsonwebtoken paketini içe aktarın:

const jwt = require('jsonwebtoken');

3. JWT Token Oluşturma

Token'ı jwt.sign metodu ile oluşturacağız. Bu metod üç parametre gerektirir: payload (yük), secret key (gizli anahtar) ve opsiyonel yapılandırma ayarları.

3.1. Payload Tanımlama

Payload, token'da yer almak istediğiniz verilerdir. Genellikle bu veriler kullanıcı ID'si ve email olacaktır. Kullanıcı kayıt edildikten sonra bu bilgileri alın:

const user = await User.create({ email, password });
const payload = {
    userId: user._id,
    email: user.email
};

3.2. Secret Key Tanımlama

Secret key, token'ı imzalamak ve daha sonra doğrulamak için kullanılır. Bu anahtar güvenli bir şekilde saklanmalı ve kaynak dosyalarınızda hard-code edilmemelidir. Hassas bilgileri yönetmek için çevre değişkenlerini kullanın.

3.3. Çevre Değişkenlerini Ayarlama

.env dosyanızda yeni bir değişken ekleyin:

TOKEN_KEY=your-strong-secret-key

your-strong-secret-key ifadesini güvenli bir şekilde oluşturulmuş bir dize ile değiştirin. Güçlü bir rastgele anahtar oluşturmak için çevrimiçi araçları kullanabilirsiniz.

3.4. Token Oluşturma

sign metodunu çağırarak token'ı oluşturun:

const token = jwt.sign(payload, process.env.TOKEN_KEY, { expiresIn: '8h' });

Bu token 8 saat sonra geçerliliğini yitirecek ve kullanıcı yeni bir token almak için tekrar giriş yapmak zorunda kalacak.

4. Token'ı Yanıtta Döndürme

Token'ı ve diğer kullanıcı bilgilerini istemciye geri gönderin. Bu genellikle yanıt nesnesinde yapılır:

res.status(201).json({
    message: 'User created successfully',
    user: {
        email: user.email,
        username: user.username,
        token: token
    }
});

5. Token'ı Sonraki İsteklerde Doğrulama

Güvenli rotalar için, isteğin kimlik doğrulamasının yapılmasını sağlamak amacıyla token'ı doğrulayın. Token doğrulama işlemini gerçekleştiren bir middleware fonksiyonu oluşturun:

const verifyToken = (req, res, next) => {
    const token = req.headers['authorization'];
    if (!token) return res.status(403).send('A token is required for authentication');

    try {
        const decoded = jwt.verify(token, process.env.TOKEN_KEY);
        req.user = decoded;
    } catch (err) {
        return res.status(401).send('Invalid Token');
    }
    return next();
};

Kimlik doğrulaması gerektiren rotalara bu middleware fonksiyonunu uygulayın:

app.get('/secure-route', verifyToken, (req, res) => {
    res.send('This is a secure route');
});

Token Doğrulama Middleware (middleware/verifyToken.js)

const jwt = require('jsonwebtoken');

const verifyToken = (req, res, next) => {
    const token = req.headers['authorization'];
    if (!token) return res.status(403).send('A token is required for authentication');

    try {
        const decoded = jwt.verify(token, process.env.TOKEN_KEY);
        req.user = decoded;
    } catch (err) {
        return res.status(401).send('Invalid Token');
    }
    return next();
};

module.exports = verifyToken;

2. Kullanıcı Kayıt Mantığı

2.1. Mevcut Kullanıcı Kontrolü

Kayıt mantığımız, verilen e-posta ile bir kullanıcının zaten veritabanında olup olmadığını kontrol etmelidir. E-posta zaten kullanımda ise, sunucu 409 Conflict durum kodu döndürmelidir.

Kod Uygulaması:

const registerUser = async (req, res) => {
    const { email, password } = req.body;

    // Kullanıcının zaten mevcut olup olmadığını kontrol et
    const existingUser = await User.findOne({ email });
    if (existingUser) {
        return res.status(409).json({ message: "Email is already in use" });
    }

    // Şifreyi şifrele
    const hashedPassword = await bcrypt.hash(password, 10);

    // Yeni kullanıcı oluştur
    const newUser = new User({
        email,
        password: hashedPassword,
    });

    await newUser.save();

    // Token oluştur
    const token = jwt.sign({ userId: newUser._id }, process.env.JWT_SECRET, {
        expiresIn: '1h'
    });

    res.status(201).json({ token, user: newUser });
};

Giriş mantığı, kullanıcının e-posta ve şifresini doğrular. Bilgiler doğruysa, bir token oluşturulur ve kullanıcıya geri gönderilir.

Kod Uygulaması:

const loginUser = async (req, res) => {
    const { email, password } = req.body;

    // Kullanıcının mevcut olup olmadığını kontrol et
    const user = await User.findOne({ email });
    if (!user) {
        return res.status(401).json({ message: "Invalid email or password" });
    }

    // Şifreleri karşılaştır
    const isPasswordValid = await bcrypt.compare(password, user.password);
    if (!isPasswordValid) {
        return res.status(401).json({ message: "Invalid email or password" });
    }

    // Token oluştur
    const token = jwt.sign({ userId: user._id }, process.env.JWT_SECRET, {
        expiresIn: '1h'
    });

    res.status(200).json({ token, user });
};

5. Hata Yönetimi ve Güvenlik

5.1. Şifre Şifreleme

Şifrelerin veritabanına kaydedilmeden önce bcrypt kullanılarak şifrelendiğinden emin olun. Bu, kullanıcı kimlik bilgilerini korumak için ek bir güvenlik katmanı ekler.

5.2. Token Oluşturma

Kullanıcıları kimlik doğrulamak için JSON Web Token (JWT) oluşturun. JWT sırrını ortam değişkenlerinde güvenli bir şekilde saklayın.

5.3. Uç Durumları Yönetme

  • Kayıt sırasında geçersiz e-posta formatlarının reddedildiğinden emin olun.
  • Giriş endpoint'lerinde brute-force saldırılarına karşı koruma sağlamak için rate limiting uygulayın.
  • SQL enjeksiyon ve diğer saldırıları önlemek için giriş verilerini doğrulayın ve temizleyin.

Giriş Fonksiyonelliği Uygulama

1. Giriş Rotasını Tanımla

Öncelikle, giriş isteklerini işlemek için bir rota oluşturun. Bu rota bir e-posta ve şifre alacak, kimlik bilgilerini doğrulayacak ve kimlik doğrulaması başarılı olursa bir token döndürecektir.

const express = require('express');
const router = express.Router();
const User = require('../models/User'); // User modelinin tanımlı olduğu varsayılıyor
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

// Giriş rotası
router.post('/login', async (req, res) => {
    const { email, password } = req.body;

    // Girdi doğrulaması
    if (!email || !password) {
        return res.status(400).json({ message: 'Email ve şifre gereklidir' });
    }

    try {
        // E-posta ile kullanıcıyı bul
        const user = await User.findOne({ email });
        if (!user) {
            return res.status(400).json({ message: 'Geçersiz kimlik bilgileri, lütfen tekrar deneyin' });
        }

        // Şifreleri karşılaştır
        const isMatch = await bcrypt.compare(password, user.password);
        if (!isMatch) {
            return res.status(400).json({ message: 'Geçersiz kimlik bilgileri, lütfen tekrar deneyin' });
        }

        // JWT token oluştur
        const token = jwt.sign(
            { userId: user._id, email: user.email },
            process.env.JWT_SECRET,
            { expiresIn: '1h' }
        );

        // Token ve kullanıcı detayları ile yanıt gönder
        res.status(200).json({
            token,
            user: {
                id: user._id,
                email: user.email,
                username: user.username
            }
        });

    } catch (error) {
        res.status(500).json({ message: 'Sunucu hatası, lütfen daha sonra tekrar deneyin' });
    }
});

module.exports = router;

            { userId: user._id, email: user.email },
            process.env.JWT_SECRET,

        );

            token,

                id: user._id,
                email: user.email,

    }
});
